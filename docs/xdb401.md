Arduino code example:

```
#include <Wire.h>

const float FULLSCALE_PRESSURE = 0.5;  // MPa
const uint8_t I2C_ADDRESS = 0x7F;

#define FILTER_SAMPLES 5  // Настройки фильтра
float pressureHistory[FILTER_SAMPLES];
int historyIndex = 0;
float zeroOffset = 0.0;

void setup() {
  Serial.begin(115200);
  Wire.begin();
  Wire.setClock(100000);
  for (int i = 0; i < FILTER_SAMPLES; i++) {  // Инициализация фильтра
    pressureHistory[i] = 0;
  }
  Serial.println();
  Serial.println("========================");
  Serial.println("Датчик давления");
  Serial.println("========================");
  Serial.println("Давление (бар)\tТемпература (°C)");
  Serial.println("------------------------");
  calibrateZero();  // Автокалибровка при запуске
}

void loop() {
  float pressure, temperature;
  readSensorData(pressure, temperature);  // Чтение и обработка данных
  float filteredPressure = applyFilter(pressure);
  float calibratedPressure = filteredPressure - zeroOffset;
  float pressureBar = calibratedPressure * 10.0;  // 1 MPa = 10 bar // Конвертация в бары и вывод
  Serial.print(pressureBar, 2);
  Serial.print(" бар\t\t");
  Serial.print(temperature, 1);
  Serial.println(" °C");
  delay(1000);
}

void readSensorData(float &pressure, float &temperature) {  // Запуск измерения
  Wire.beginTransmission(I2C_ADDRESS);
  Wire.write(0x30);
  Wire.write(0x0A);
  Wire.endTransmission();
  delay(50);
  Wire.beginTransmission(I2C_ADDRESS);  // Чтение давления (3 байта)
  Wire.write(0x06);
  Wire.endTransmission(false);
  Wire.requestFrom((uint8_t)I2C_ADDRESS, (uint8_t)3);  // Явное указание типов
  uint8_t pressureData[3];
  for (int i = 0; i < 3; i++) {
    if (Wire.available()) {
      pressureData[i] = Wire.read();
    }
  }
  Wire.beginTransmission(I2C_ADDRESS);  // Чтение температуры (2 байта)
  Wire.write(0x09);
  Wire.endTransmission(false);
  Wire.requestFrom((uint8_t)I2C_ADDRESS, (uint8_t)2);  // Явное указание типов
  uint8_t tempData[2];
  for (int i = 0; i < 2; i++) {
    if (Wire.available()) {
      tempData[i] = Wire.read();
    }
  }
  uint32_t pressureRAW = ((uint32_t)pressureData[0] << 16) | ((uint32_t)pressureData[1] << 8) | pressureData[2];  // Конвертация
  uint16_t temperatureRAW = ((uint16_t)tempData[0] << 8) | tempData[1];
  pressure = convertPressure(pressureRAW);
  temperature = convertTemperature(temperatureRAW);
}

float applyFilter(float newValue) {
  pressureHistory[historyIndex] = newValue;
  historyIndex = (historyIndex + 1) % FILTER_SAMPLES;
  float sum = 0;
  for (int i = 0; i < FILTER_SAMPLES; i++) {
    sum += pressureHistory[i];
  }
  return sum / FILTER_SAMPLES;
}

void calibrateZero() {
  const int samples = 8;
  float sum = 0;
  for (int i = 0; i < samples; i++) {
    float pressure, temperature;
    readSensorData(pressure, temperature);
    float filtered = applyFilter(pressure);
    sum += filtered;
    delay(200);
  }
  zeroOffset = sum / samples;
}

float convertPressure(uint32_t adcValue) {
  float normalizedPressure;
  if (adcValue > 8388608) {
    normalizedPressure = (adcValue - 16777216.0) / 8388608.0;
  } else {
    normalizedPressure = adcValue / 8388608.0;
  }
  return normalizedPressure * FULLSCALE_PRESSURE;
}

float convertTemperature(uint16_t adcValue) {
  if (adcValue > 32768) {
    return (adcValue - 65536.0) / 256.0;
  } else {
    return adcValue / 256.0;
  }
}
```
